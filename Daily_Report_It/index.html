🔗 วิธีการเชื่อมต่อ Google Sheet

  1. สร้าง Google Sheet ใหม่ เพื่อเก็บข้อมูล Daily Report
  2. เปิด Apps Script ไปที่ Extensions → Apps Script
  3. คัดลอกโค้ด Google Apps Script (ดูรายละเอียดด้านล่าง)
  4. ตั้งค่า Sheet ID และ Folder ID ในค่าตัวแปร
  5. Deploy เป็น Web App และคัดลอก Web App URL
  6. วางลิงค์ Web App URL ไปด้านล่าง

📋 Google Apps Script Code

ต่อไปนี้คือโค้ด Google Apps Script ที่ต้องวาง ไปที่ Extensions → Apps Script 

// ตั้งค่าตัวแปร
const SHEET_ID = "YOUR_SHEET_ID";
const FOLDER_ID = "YOUR_FOLDER_ID";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
    
    // แปลงวันที่
    const dateObj = new Date(data.date);
    const thaiDate = Utilities.formatDate(dateObj, "Asia/Bangkok", "dd/MM/yyyy");
    
    // เพิ่มแถวใหม่
    sheet.appendRow([
      thaiDate,
      data.title,
      data.details,
      data.status,
      data.remark,
      data.imageUrl || ""
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      message: "บันทึกสำเร็จ"
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function uploadImage(imageData) {
  const folder = DriveApp.getFolderById(FOLDER_ID);
  const blob = Utilities.newBlob(imageData, "image/jpeg", "report_image.jpg");
  const file = folder.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE, DriveApp.Permission.VIEW);
  return file.getUrl();
}
