📚 วิธีการตั้งค่า

1. สร้าง Google Sheet

• ไปที่ Google Drive
• New → Google Sheets → Blank
• ตั้งชื่อ "IT Daily Report"
• สร้างหัวข้อ: วันที่ | หัวข้องาน | รายละเอียด | สถานะ | หมายเหตุ

2. สร้าง Apps Script

• ใน Google Sheet → Extensions → Apps Script
• วาง Code ด้านล่าง
• แทนที่ YOUR_SHEET_ID ด้วย ID ของ Sheet
• บันทึก (Ctrl+S)

3. Deploy

• Deploy → New Deployment
• Type: "Web app"
• Execute as: Email ของคุณ
• Who has access: "Anyone"
• Deploy
• คัดลอก Web App URL

4. Apps Script Code

const SHEET_ID = "YOUR_SHEET_ID";
const SHEET_NAME = "Sheet1";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName(SHEET_NAME);
    
    sheet.appendRow([
      data.date,
      data.title,
      data.details,
      data.status,
      data.remark
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      message: "บันทึกสำเร็จ"
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}
