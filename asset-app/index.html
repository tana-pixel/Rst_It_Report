function doGet(e) { return main(e); } function doPost(e) { return main(e); } function main(e) { try { var param = e && e.parameter ? e.parameter : {}; var action = param.action || ''; var cb = param.cb || ''; // ── test: ตรวจว่าเชื่อมต่อได้ ────────────────────── if (action === 'test') { var names = SpreadsheetApp.getActiveSpreadsheet() .getSheets().map(function(s){ return s.getName(); }); return out({ status:'ok', msg:'connected', sheets:names }, cb); } // ── หา Sheet ──────────────────────────────────────── var sheetName = param.sheet || ''; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = sheetName ? ss.getSheetByName(sheetName) : ss.getActiveSheet(); if (!sheet) { return out({ status:'error', msg:'Sheet not found: ' + sheetName }, cb); } // ── getAll: ดึงข้อมูลทั้งหมด (ไม่จำกัด row) ──────── if (action === 'getAll') { var lastRow = sheet.getLastRow(); if (lastRow < 2) { return out({ status:'ok', rows:[] }, cb); // ยังไม่มีข้อมูล } var lastCol = Math.max(sheet.getLastColumn(), 1); var vals = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues(); var rows = []; for (var i = 0; i < vals.length; i++) { var first = vals[i][0]; if (first !== '' && first !== null && first !== undefined) { rows.push(vals[i]); } } return out({ status:'ok', rows:rows }, cb); } // ── add: เพิ่มแถวใหม่ ──────────────────────────────── if (action === 'add') { var row = JSON.parse(param.row); sheet.appendRow(row); SpreadsheetApp.flush(); return out({ status:'ok' }, cb); } // ── update: แก้ไขแถวที่ key ตรง ───────────────────── if (action === 'update') { var key = String(param.key); var row = JSON.parse(param.row); var data = sheet.getDataRange().getValues(); for (var i = data.length - 1; i >= 1; i--) { if (String(data[i][0]) === key) { sheet.getRange(i + 1, 1, 1, row.length).setValues([row]); SpreadsheetApp.flush(); return out({ status:'ok' }, cb); } } return out({ status:'error', msg:'key not found: ' + key }, cb); } // ── delete: ลบแถวที่ key ตรง ───────────────────────── if (action === 'delete') { var key = String(param.key); var data = sheet.getDataRange().getValues(); for (var i = data.length - 1; i >= 1; i--) { if (String(data[i][0]) === key) { sheet.deleteRow(i + 1); SpreadsheetApp.flush(); return out({ status:'ok' }, cb); } } return out({ status:'error', msg:'key not found: ' + key }, cb); } return out({ status:'error', msg:'unknown action: ' + action }, cb); } catch(err) { return out({ status:'error', msg: err.message || String(err) }, e && e.parameter ? e.parameter.cb || '' : ''); } } // helper: สร้าง ContentService output (JSON หรือ JSONP) function out(obj, cb) { var json = JSON.stringify(obj); var body = cb ? cb + '(' + json + ')' : json; var mime = cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON; return ContentService.createTextOutput(body).setMimeType(mime); }

💬 แจ้งเตือน LINE (Messaging API)

รับแจ้งเตือนไลน์เมื่อมีการสร้าง Ticket / เพิ่มทรัพย์สิน / เบิกโอน

หา User ID ได้จาก LINE Official Account Manager → Chat → คลิกชื่อผู้ใช้

⚠️ เกี่ยวกับ CORS — ต้องใช้ Apps Script Proxy

Browser ไม่สามารถส่งตรงไป api.line.me ได้ (CORS) ระบบจะส่งผ่าน Apps Script ที่ตั้งค่าไว้ใน Ticket URL โดยอัตโนมัติ
ต้องเพิ่มโค้ด sendLine() ใน Apps Script ของคุณ (ดูคำแนะนำด้านล่าง)

// เพิ่มฟังก์ชันนี้ใน Apps Script เดิมของคุณ (IT Ticket หรือ IT Asset) // แล้ว Deploy ใหม่ function sendLine(token, userId, msg) { var url = 'https://api.line.me/v2/bot/message/push'; var payload = JSON.stringify({ to: userId, messages: [{ type: 'text', text: msg }] }); var options = { method: 'post', contentType: 'application/json', headers: { 'Authorization': 'Bearer ' + token }, payload: payload, muteHttpExceptions: true }; try { var res = UrlFetchApp.fetch(url, options); return res.getResponseCode() === 200; } catch(e) { return false; } } // แก้ไขฟังก์ชัน main() เดิม — เพิ่ม block นี้ต่อท้าย action อื่นๆ if (action === 'sendLine') { var ok = sendLine(param.token, param.userId, param.msg); return out({ status: ok ? 'ok' : 'error' }, cb); }
