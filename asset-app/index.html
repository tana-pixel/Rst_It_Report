ลบ code เดิม → วาง code ด้านล่าง → Save (Ctrl+S)
function doGet(e) { return handleReq(e); } function doPost(e) { return handleReq(e); } function handleReq(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var p = e.parameter; if (p.action === 'add') { sheet.appendRow([p.id,p.type,p.brand,p.model,p.user,p.dept,p.status,p.date,p.serial,p.price,p.note,new Date().toLocaleDateString('th-TH')]); } else if (p.action === 'update') { var v = sheet.getDataRange().getValues(); for (var i=1;i=1;i--) { if (v[i][0]==p.id) { sheet.deleteRow(i+1); break; } } } var res = JSON.stringify({status:'ok'}); return ContentService.createTextOutput(p.cb ? p.cb+'('+res+')' : res) .setMimeType(p.cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON); } catch(err) { var res = JSON.stringify({status:'error',msg:err.toString()}); return ContentService.createTextOutput(e.parameter.cb ? e.parameter.cb+'('+res+')' : res) .setMimeType(ContentService.MimeType.JAVASCRIPT); } }
